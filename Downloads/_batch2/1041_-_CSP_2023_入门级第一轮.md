	int min(int x, int y, int z) {

	int edit_dist_dp(string str1, string str2) {

	    int m = str1.length();

	    int n = str2.length();

	    vector<vector<int>> dp(m + 1, vector<int>(n + 1));

	    for (int i = 0; i <= m; i++) {

	        for (int j = 0; j <= n; j++) {

	            if (i == 0)
