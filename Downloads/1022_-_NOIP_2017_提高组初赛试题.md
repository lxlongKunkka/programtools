	#include <iostream>

	void mergesort(int l, int r)

	{

	    if (l == r)

	        return;

	    int mid = (l + r) / 2;

	    int p = l;

	    int i = l;

	    int j = mid + 1;

	    mergesort(l, mid);

	    mergesort(mid + 1, r);

	    while (i <= mid && j <= r)

	    {

	        if (a[j] < a[i])

	        {

	            s += mid - i + 1;

	            t[p] = a[j];

	            p++;

	            j++;

	        }

	        else

	        {

	            t[p] = a[i];

	            p++;

	            i++;

	        }

	    }

	    {

	        t[p] = a[i];

	        p++;

	        i++;

	    }

	    while (j <= r)

	    {

	        t[p] = a[j];

	        p++;

	        j++;

	    }

	    for (i = l; i <= r; i++)

	        a[i] = t[i];
